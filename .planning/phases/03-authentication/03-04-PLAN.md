    Create `my-app/components/auth/Guard.tsx`:
    ```tsx
    'use client'
    import { useEffect, useState } from 'react'
    import { useRouter, usePathname } from 'next/navigation'
    import { doc, getDoc } from 'firebase/firestore'
    import { db } from '@/lib/firebase/client'
    import { useAuth } from '@/components/auth/AuthProvider'
    import { Skeleton } from '@/components/ui/skeleton'

    function buildSignInRedirect(pathname: string): string {
      // Only relative, same-origin paths are honored — prevents open-redirect via
      // a crafted `redirect` value that points to an external host (T-03-10).
      const safePath = pathname.startsWith('/') && !pathname.startsWith('//') ? pathname : '/'
      return `/sign-in?redirect=${encodeURIComponent(safePath)}`
    }

    export function Guard({
      children,
      requireVerified = false,
    }: {
      children: React.ReactNode
      requireVerified?: boolean
    }) {
      const { user, loading, emailVerified } = useAuth()
      const router = useRouter()
      const pathname = usePathname()
      const [checkingUsername, setCheckingUsername] = useState(true)
      const [hasUsername, setHasUsername] = useState(false)

      useEffect(() => {
        if (loading) return

        if (!user) {
          router.replace(buildSignInRedirect(pathname))
          return
        }

        if (requireVerified && !emailVerified) {
          router.replace('/verify-email')
          return
        }

        let cancelled = false
        getDoc(doc(db, 'users', user.uid)).then((snap) => {
          if (cancelled) return
          const username = snap.data()?.username
          if (!username) {
            router.replace('/onboarding/username')
          } else {
            setHasUsername(true)
          }
          setCheckingUsername(false)
        })

        return () => {
          cancelled = true
        }
      }, [loading, user, emailVerified, requireVerified, pathname, router])

      if (loading || checkingUsername || !user || !hasUsername) {
        return (
          <div className="flex min-h-screen flex-col items-center justify-center gap-3 p-8">
            <Skeleton className="h-8 w-48" />
            <Skeleton className="h-4 w-64" />
            <Skeleton className="h-4 w-56" />
          </div>
        )
      }

      return <>{children}</>
    }
    ```
    Note the sequencing per RESEARCH.md Pitfall 4: (1) loading check first, (2) unauthenticated redirect second, (3) `requireVerified` check third — BEFORE the username Firestore read, (4) username check last. This ordering ensures a freshly-signed-up unverified user is never sent to onboarding before being sent to verify-email, and a verified-but-no-username user reaches onboarding correctly.

    Also export `buildSignInRedirect` is intentionally kept module-private (not exported) — it's an internal helper, not part of Guard's public contract.

    ```tsx
    'use client'
    import { useState } from 'react'
    import { useRouter } from 'next/navigation'
    import { useForm, useWatch } from 'react-hook-form'
    import { zodResolver } from '@hookform/resolvers/zod'
    import { doc, updateDoc } from 'firebase/firestore'
    import { Loader2 } from 'lucide-react'
    import { db } from '@/lib/firebase/client'
    import { useAuth } from '@/components/auth/AuthProvider'
    import { Guard } from '@/components/auth/Guard'
    import { usernameSchema, type UsernameInput } from '@/lib/firebase/auth-schemas'
    import { Card, CardContent, CardHeader, CardTitle } from '@/components/ui/card'
    import { Button } from '@/components/ui/button'
    import { Input } from '@/components/ui/input'
    import { Label } from '@/components/ui/label'

    function UsernameForm() {
      const { user } = useAuth()
      const router = useRouter()
      const [submitError, setSubmitError] = useState<string | null>(null)
      const {
        register,
        handleSubmit,
        control,
        formState: { errors, isSubmitting },
      } = useForm<UsernameInput>({ resolver: zodResolver(usernameSchema) })
      const liveUsername = useWatch({ control, name: 'username' }) || ''

      async function onSubmit(values: UsernameInput) {
        if (!user) return
        setSubmitError(null)
        try {
          await updateDoc(doc(db, 'users', user.uid), { username: values.username })
          router.push('/')
        } catch {
          setSubmitError('Could not save username. Please try again.')
        }
      }

      return (
        <Card>
          <CardHeader>
            <CardTitle className="text-3xl font-semibold">Choose a username</CardTitle>
          </CardHeader>
          <CardContent className="flex flex-col gap-4">
            <p className="text-sm text-muted-foreground">
              This will be your public profile URL: easyappointment.app/u/{liveUsername || '...'}
            </p>
            <form onSubmit={handleSubmit(onSubmit)} className="flex flex-col gap-4">
              {submitError && <p className="text-sm text-destructive">{submitError}</p>}
              <div className="flex flex-col gap-2">
                <Label htmlFor="username">Username</Label>
                <Input id="username" autoComplete="off" {...register('username')} />
                {errors.username && <p className="text-sm text-destructive">{errors.username.message}</p>}
              </div>
              <Button type="submit" disabled={isSubmitting} className="w-full">
                {isSubmitting && <Loader2 className="animate-spin" />}
                Continue
              </Button>
            </form>
          </CardContent>
        </Card>
      )
    }

    export default function UsernameOnboardingPage() {
      return (
        <Guard requireVerified={false}>
          <UsernameForm />
        </Guard>
      )
    }
    ```
    Note: wrapping this page in `<Guard>` creates a self-referential edge case — `Guard` redirects TO `/onboarding/username` when username is missing, and this very page is `/onboarding/username`. This is safe because `Guard`'s username check only redirects when `!username`; once the page's own form successfully writes a username and calls `router.push('/')`, the next navigation to a Guard-wrapped page will find `hasUsername` true and render normally. `Guard` itself does not loop on this page because the check only fires once per mount and the page navigates away on success — confirm no infinite redirect loop by manually testing: sign up, verify email, land on this page once (not repeatedly bouncing).

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|--------------|
| `?redirect=` query param → `router.replace()` | User-controllable input from the URL crosses into a navigation call. |
| Client-side Guard check → actual data access | Guard is UX-only; real enforcement is Firestore Security Rules (Phase 2). |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|------------------|
| T-03-10 | Tampering / Spoofing (open redirect) | `Guard.tsx` sign-in redirect construction | mitigate | `buildSignInRedirect` validates the path starts with a single `/` and rejects `//`-prefixed values (protocol-relative external redirect) before using it in `router.replace()`, per RESEARCH.md Security Domain's explicit flag on this exact pattern. |
| T-03-11 | Elevation of Privilege | Client-side-only Guard bypass via direct Firestore/API calls | accept | `Guard.tsx` is explicitly documented (RESEARCH.md Anti-Patterns) as a UX redirect layer only; actual enforcement is Firestore Security Rules from Phase 2 (`users/{userId}` requires matching uid) plus any future server-side Route Handler checks in later phases. This phase does not add new server-side enforcement beyond what Phase 2 already provides, which is the correct boundary per Next.js's own authentication guidance. |
| T-03-12 | Tampering | `/onboarding/username` Firestore write | mitigate | `updateDoc` targets `users/{user.uid}` using the current authenticated user's own uid (from `useAuth()`), satisfying Phase 2's existing security rule (`request.auth.uid == userId`) — no new rule needed. |
</threat_model>
