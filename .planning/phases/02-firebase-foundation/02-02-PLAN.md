```typescript
// Firebase client SDK singleton
// Import this file in "use client" components and hooks only.
// Never import from Route Handlers (use admin.ts instead).
//
// Source: https://firebase.google.com/docs/web/setup
// Source: https://firebase.google.com/docs/emulator-suite/connect_firestore

import { initializeApp, getApps, getApp } from 'firebase/app'
import { getAuth, connectAuthEmulator } from 'firebase/auth'
import { getFirestore, connectFirestoreEmulator } from 'firebase/firestore'
import { getStorage, connectStorageEmulator } from 'firebase/storage'

const firebaseConfig = {
  apiKey: process.env.NEXT_PUBLIC_FIREBASE_API_KEY!,
  authDomain: process.env.NEXT_PUBLIC_FIREBASE_AUTH_DOMAIN!,
  projectId: process.env.NEXT_PUBLIC_FIREBASE_PROJECT_ID!,
  storageBucket: process.env.NEXT_PUBLIC_FIREBASE_STORAGE_BUCKET!,
  messagingSenderId: process.env.NEXT_PUBLIC_FIREBASE_MESSAGING_SENDER_ID!,
  appId: process.env.NEXT_PUBLIC_FIREBASE_APP_ID!,
}

// getApps().length guard prevents "Firebase App named '[DEFAULT]' already exists"
// on Next.js hot reload and module re-import.
const app = getApps().length ? getApp() : initializeApp(firebaseConfig)

export const auth = getAuth(app)
export const db = getFirestore(app)
export const storage = getStorage(app)

// Emulator connections — must happen before any SDK operation on these instances.
// The module singleton ensures these run at most once per process.
// connectXxxEmulator throws on second call — the module cache prevents double-calls.
if (process.env.NEXT_PUBLIC_USE_FIREBASE_EMULATOR === 'true') {
  connectAuthEmulator(auth, 'http://127.0.0.1:9099', { disableWarnings: true })
  connectFirestoreEmulator(db, '127.0.0.1', 8080)
  connectStorageEmulator(storage, '127.0.0.1', 9199)
}
```

// TypeScript interfaces for all 5 Firestore collections.
// These are the canonical document shapes — every Phase 3–12 plan imports from here.
// Schema source: .planning/phases/02-firebase-foundation/02-CONTEXT.md §D-02
//
// Timestamp: use firebase/firestore Timestamp (client) or firebase-admin/firestore Timestamp (admin).
// This file uses the CLIENT SDK Timestamp for use in "use client" components.
// Admin SDK code should cast or use firebase-admin/firestore Timestamp directly.

import { Timestamp, FieldValue } from 'firebase/firestore'

// ---- Shared enums / union types ----

export type SlotStatus = 'available' | 'pending' | 'booked' | 'unavailable'
export type BookingStatus = 'confirmed' | 'cancelled' | 'pending_payment'
export type PaymentStatus = 'not_required' | 'paid' | 'pending' | 'failed'
export type CalendarStatus = 'draft' | 'published' | 'archived'
export type CalendarVisibility = 'public' | 'invite-only'

// ---- Collection: users/{uid} ----

export interface UserDoc {
  uid: string
  email: string
  displayName: string
  username: string          // used for /u/[username] (PROF-04)
  bio: string
  photoURL: string | null
  emailVerified: boolean
  createdAt: Timestamp
  updatedAt: Timestamp
}

// ---- Collection: calendars/{calendarId} ----

export interface DailySlotConfig {
  startTime: string         // "09:00" HH:mm local-agnostic
  endTime: string           // "17:00"
  durationMinutes: number
}

export interface PaymentConfig {
  amount: number            // in cents
  currency: string          // e.g. "usd"
  description: string
}

export interface CalendarDoc {
  id: string
  creatorId: string         // Firebase Auth uid
  title: string
  status: CalendarStatus
  visibility: CalendarVisibility
  startDate: string         // UTC date "2026-05-19" — matches PKG-02 contract
  endDate: string           // UTC date "2026-05-30"
  dailySlotConfig: DailySlotConfig
  invitedEmails: string[]   // whitelist for invite-only calendars (INV-01)
  isPaid: boolean
  paymentConfig: PaymentConfig | null   // null when isPaid is false
  smtpConfigEncrypted: string | null    // AES-256 encrypted blob (SMTP-03); null = platform default
  googleCalendarConnected: boolean
  createdAt: Timestamp
  updatedAt: Timestamp
}

// ---- Collection: slots/{slotId} ----

export interface SlotDoc {
  id: string
  calendarId: string
  creatorId: string         // denormalized from calendar — required for security rules without cross-doc reads
  date: string              // UTC date key "2026-05-19" (Phase 1 D-02)
  startUtc: string          // ISO 8601 "2026-05-19T09:00:00Z" (Phase 1 D-01)
  endUtc: string            // ISO 8601 "2026-05-19T10:00:00Z"
  status: SlotStatus
  bookingId: string | null
  pendingExpiresAt: Timestamp | null    // PAY-04: auto-reset after 30 min
  createdAt: Timestamp
  updatedAt: Timestamp
}

// Write variant: timestamps may be server FieldValue on creation
export type SlotDocWrite = Omit<SlotDoc, 'createdAt' | 'updatedAt'> & {
  createdAt: Timestamp | FieldValue
  updatedAt: Timestamp | FieldValue
}

// ---- Collection: bookings/{bookingId} ----

export interface BookingDoc {
  id: string
  slotId: string
  calendarId: string
  creatorId: string         // denormalized — for dashboard queries (DASH-01, DASH-02)
  bookerEmail: string
  bookerUid: string | null  // null for guest bookers
  status: BookingStatus
  stripeSessionId: string | null
  stripeEventId: string | null   // idempotency key (PAY-06)
  paymentStatus: PaymentStatus
  createdAt: Timestamp
  updatedAt: Timestamp
}

// ---- Collection: invites/{inviteId} — OTP verification tokens ----

export interface InviteDoc {
  id: string
  calendarId: string
  email: string             // the email the OTP was sent to
  token: string             // 6-digit code (SHA-256 HMAC hashed — see RESEARCH.md §OTP)
  used: boolean             // set atomically to true on first use (INV-04)
  expiresAt: Timestamp      // 15 minutes from creation (INV-04)
  createdAt: Timestamp
}
```

// Firebase Admin SDK singleton — SERVER ONLY.
// This file imports 'server-only' which causes a build error if it is ever
// imported from a "use client" component or the client bundle.
//
// Import adminDb and adminAuth only in:
//   - app/api/**/*.ts (Route Handlers)
//   - Server Actions (if used in future phases)
//   - scripts/ (seed scripts, admin utilities)
//
// Source: https://firebase.google.com/docs/admin/setup
// Credential pattern: benmvp.com/blog/initializing-firebase-admin-node-sdk-env-vars/

import 'server-only'
import { getApps, initializeApp, type App } from 'firebase-admin/app'
import { credential } from 'firebase-admin'
import { getFirestore } from 'firebase-admin/firestore'
import { getAuth } from 'firebase-admin/auth'

function getAdminApp(): App {
  // getApps() guard prevents "Firebase App named '[DEFAULT]' already exists"
  // across multiple imports in the same process (e.g. multiple Route Handlers).
  if (getApps().length > 0) {
    return getApps()[0]
  }

  return initializeApp({
    credential: credential.cert({
      projectId: process.env.FIREBASE_PROJECT_ID!,
      clientEmail: process.env.FIREBASE_CLIENT_EMAIL!,
      // FIREBASE_PRIVATE_KEY is stored with literal \n in env vars (Vercel, .env.local).
      // .replace(/\\n/g, '\n') converts them back to real newlines for the PEM parser.
      privateKey: process.env.FIREBASE_PRIVATE_KEY!.replace(/\\n/g, '\n'),
    }),
  })
}

const adminApp = getAdminApp()

export const adminDb = getFirestore(adminApp)
export const adminAuth = getAuth(adminApp)
```

CRITICAL checks before saving:
1. The file MUST start with `import 'server-only'` as the first import statement.
2. The `credential` import is from `'firebase-admin'` (the compat namespace export), NOT from `'firebase-admin/app'`.
3. `getFirestore` and `getAuth` are imported from their respective subpackages (`'firebase-admin/firestore'`, `'firebase-admin/auth'`).
4. `getApps`, `initializeApp`, `App` are imported from `'firebase-admin/app'`.
5. The private key replace pattern is exactly `.replace(/\\n/g, '\n')` — note the double-backslash in the regex literal which matches a literal `\n` two-character sequence.

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| client bundle → Firebase project | client.ts config values are public by design; security enforced by Firestore rules |
| Route Handler → Admin SDK | admin.ts is the server trust boundary; only it can write bookings/invites |
| env vars → admin.ts credential | FIREBASE_PRIVATE_KEY must never appear in client bundle or logs |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-2-02-01 | Information Disclosure | `admin.ts` imported in client bundle | mitigate | `import 'server-only'` at top of admin.ts; Next.js throws build error if violated |
| T-2-02-02 | Information Disclosure | `FIREBASE_PRIVATE_KEY` logged in console | mitigate | admin.ts never logs credentials; private key is used only in `credential.cert()` call |
| T-2-02-03 | Tampering | Client SDK bypasses Admin SDK trust boundary | mitigate | Firestore security rules (Plan 03) deny all client writes to bookings/invites; Admin SDK is the only writer |
| T-2-02-04 | Tampering | `connectXxxEmulator` called in production | accept | Guard is `process.env.NEXT_PUBLIC_USE_FIREBASE_EMULATOR === 'true'`; production builds have this unset; no risk |
</threat_model>
