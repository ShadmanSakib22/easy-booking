```typescript
/**
 * Emulator Seed Script
 * Phase 2 — Firebase Foundation (D-07)
 *
 * Seeds the Firebase emulator with test data for development.
 * Creates: 1 test user, 1 published public calendar, 5 available slots.
 *
 * Prerequisites:
 *   Firebase emulators must be running:
 *     firebase emulators:start --project demo-easyappointment
 *
 * Run via:
 *   pnpm seed
 *   (sets FIRESTORE_EMULATOR_HOST and FIREBASE_AUTH_EMULATOR_HOST automatically)
 *
 * Or manually:
 *   FIRESTORE_EMULATOR_HOST=127.0.0.1:8080 FIREBASE_AUTH_EMULATOR_HOST=127.0.0.1:9099 tsx scripts/seed-emulator.ts
 */

import { initializeApp } from 'firebase-admin/app'
import { getFirestore, Timestamp } from 'firebase-admin/firestore'
import { getAuth } from 'firebase-admin/auth'

// Admin SDK auto-connects to emulators via env vars.
// FIRESTORE_EMULATOR_HOST and FIREBASE_AUTH_EMULATOR_HOST must be set
// before initializeApp() is called — the pnpm seed script handles this.
initializeApp({ projectId: 'demo-easyappointment' })

const db = getFirestore()
const auth = getAuth()

async function seed(): Promise<void> {
  console.log('Starting emulator seed...')

  // ── Step 1: Create test user in Auth emulator ──────────────────────────────
  let userRecord
  try {
    userRecord = await auth.createUser({
      email: 'creator@test.com',
      password: 'password123',
      displayName: 'Test Creator',
      emailVerified: true,
    })
    console.log(`Created user: ${userRecord.uid}`)
  } catch (err: unknown) {
    // If user already exists (re-running seed), get existing user instead
    if (
      typeof err === 'object' &&
      err !== null &&
      'code' in err &&
      (err as { code: string }).code === 'auth/email-already-exists'
    ) {
      userRecord = await auth.getUserByEmail('creator@test.com')
      console.log(`User already exists: ${userRecord.uid}`)
    } else {
      throw err
    }
  }

  const uid = userRecord.uid
  const now = Timestamp.now()

  // ── Step 2: Write user document ────────────────────────────────────────────
  await db.collection('users').doc(uid).set({
    uid,
    email: 'creator@test.com',
    displayName: 'Test Creator',
    username: 'testcreator',
    bio: 'Seed data test creator account.',
    photoURL: null,
    emailVerified: true,
    createdAt: now,
    updatedAt: now,
  })
  console.log(`Wrote users/${uid}`)

  // ── Step 3: Write calendar document ───────────────────────────────────────
  const calendarId = 'seed-cal-1'
  await db.collection('calendars').doc(calendarId).set({
    id: calendarId,
    creatorId: uid,
    title: 'Test Calendar',
    status: 'published',
    visibility: 'public',
    startDate: '2026-07-01',
    endDate: '2026-07-31',
    dailySlotConfig: {
      startTime: '09:00',
      endTime: '14:00',
      durationMinutes: 60,
    },
    invitedEmails: [],
    isPaid: false,
    paymentConfig: null,
    smtpConfigEncrypted: null,
    googleCalendarConnected: false,
    createdAt: now,
    updatedAt: now,
  })
  console.log(`Wrote calendars/${calendarId}`)

  // ── Step 4: Write 5 available slots ───────────────────────────────────────
  const batch = db.batch()
  for (let i = 0; i < 5; i++) {
    const slotId = `seed-slot-${i}`
    const hour = 9 + i
    const hourStr = String(hour).padStart(2, '0')
    const nextHourStr = String(hour + 1).padStart(2, '0')

    batch.set(db.collection('slots').doc(slotId), {
      id: slotId,
      calendarId,
      creatorId: uid,
      date: '2026-07-01',
      startUtc: `2026-07-01T${hourStr}:00:00Z`,
      endUtc: `2026-07-01T${nextHourStr}:00:00Z`,
      status: 'available',
      bookingId: null,
      pendingExpiresAt: null,
      createdAt: now,
      updatedAt: now,
    })
  }
  await batch.commit()
  console.log('Wrote 5 slots: seed-slot-0 through seed-slot-4')

  console.log('\nSeed complete.')
  console.log(`  Creator UID: ${uid}`)
  console.log(`  Calendar ID: ${calendarId}`)
  console.log(`  Emulator UI: http://localhost:4000`)
}

seed().catch((err) => {
  console.error('Seed failed:', err)
  process.exit(1)
})
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| .env.local → process.env | Real Firebase credentials must never be committed to git |
| seed script → emulator | Seed uses Admin SDK with no real credentials; safe for demo project |
| pnpm dev → emulator ports | Emulator traffic stays local; no production Firebase traffic in dev |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-2-05-01 | Information Disclosure | `.env.local` committed to git with real credentials | mitigate | Verify `.env.local` is in `.gitignore` before Step 1: `grep ".env.local" .gitignore` must output a line |
| T-2-05-02 | Information Disclosure | Seed script accidentally runs against production Firebase | mitigate | Seed script uses `initializeApp({ projectId: 'demo-easyappointment' })` with no credentials; Admin SDK will fail if FIRESTORE_EMULATOR_HOST is not set (won't connect to real Firestore) |
| T-2-05-03 | Denial of Service | `pnpm dev` kills both processes if emulators fail (--kill-others-on-fail) | accept | This is intentional — silent emulator failure while Next.js runs against production would be worse |
</threat_model>
