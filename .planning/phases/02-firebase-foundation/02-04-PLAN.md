- Authenticated creator tries to write booking directly → FAILS
- Authenticated non-creator, non-booker tries to read booking → FAILS

**invites collection:**
- Any authenticated user writes invite → FAILS
- Any authenticated user reads invite → FAILS

/**
 * Firestore Security Rules Test Suite
 * Phase 2 — Firebase Foundation (D-05)
 *
 * Prerequisites:
 *   - Firebase Firestore emulator running at 127.0.0.1:8080
 *   - Start with: firebase emulators:start --only firestore
 *   - Run with: cd my-app && pnpm test:rules
 *
 * API: @firebase/rules-unit-testing v5 — uses initializeTestEnvironment()
 * NOT the removed v1/v2 initializeTestApp() API.
 */

import {
  assertFails,
  assertSucceeds,
  initializeTestEnvironment,
  type RulesTestEnvironment,
} from '@firebase/rules-unit-testing'
import { doc, setDoc, getDoc } from 'firebase/firestore'
import { readFileSync } from 'fs'
import { resolve } from 'path'
import { afterAll, beforeAll, beforeEach, describe, it } from 'vitest'

let testEnv: RulesTestEnvironment

beforeAll(async () => {
  testEnv = await initializeTestEnvironment({
    projectId: 'demo-easyappointment',
    firestore: {
      // __dirname is my-app/lib/firebase/__tests__/
      // firestore.rules is at repo root: ../../../../firestore.rules
      rules: readFileSync(resolve(__dirname, '../../../../firestore.rules'), 'utf8'),
      host: '127.0.0.1',
      port: 8080,
    },
  })
})

beforeEach(async () => {
  // Clear all data between tests to prevent state leakage
  await testEnv.clearFirestore()
})

afterAll(async () => {
  await testEnv.cleanup()
})

// ── users collection ─────────────────────────────────────────────────────────

describe('users collection', () => {
  it('rejects unauthenticated write to user document', async () => {
    const unauth = testEnv.unauthenticatedContext()
    await assertFails(
      setDoc(doc(unauth.firestore(), 'users/uid-alice'), { displayName: 'Alice' })
    )
  })

  it('allows authenticated user to write their own document', async () => {
    const alice = testEnv.authenticatedContext('uid-alice')
    await assertSucceeds(
      setDoc(doc(alice.firestore(), 'users/uid-alice'), {
        uid: 'uid-alice',
        email: 'alice@example.com',
        displayName: 'Alice',
        username: 'alice',
        bio: '',
        photoURL: null,
        emailVerified: false,
        createdAt: new Date(),
        updatedAt: new Date(),
      })
    )
  })

  it('rejects write to another user document', async () => {
    const bob = testEnv.authenticatedContext('uid-bob')
    await assertFails(
      setDoc(doc(bob.firestore(), 'users/uid-alice'), { displayName: 'Hacked' })
    )
  })
})

// ── calendars collection ─────────────────────────────────────────────────────

describe('calendars collection', () => {
  it('allows unauthenticated read of published+public calendar', async () => {
    // Seed with security rules bypassed
    await testEnv.withSecurityRulesDisabled(async (ctx) => {
      await setDoc(doc(ctx.firestore(), 'calendars/cal-public'), {
        id: 'cal-public',
        creatorId: 'uid-creator',
        status: 'published',
        visibility: 'public',
        title: 'Public Calendar',
      })
    })

    const unauth = testEnv.unauthenticatedContext()
    await assertSucceeds(
      getDoc(doc(unauth.firestore(), 'calendars/cal-public'))
    )
  })

  it('rejects unauthenticated read of invite-only calendar', async () => {
    await testEnv.withSecurityRulesDisabled(async (ctx) => {
      await setDoc(doc(ctx.firestore(), 'calendars/cal-invite'), {
        id: 'cal-invite',
        creatorId: 'uid-creator',
        status: 'published',
        visibility: 'invite-only',
        title: 'Private Calendar',
      })
    })

    const unauth = testEnv.unauthenticatedContext()
    await assertFails(
      getDoc(doc(unauth.firestore(), 'calendars/cal-invite'))
    )
  })

  it('allows creator to create their own calendar', async () => {
    const creator = testEnv.authenticatedContext('uid-creator')
    await assertSucceeds(
      setDoc(doc(creator.firestore(), 'calendars/cal-new'), {
        id: 'cal-new',
        creatorId: 'uid-creator',
        status: 'draft',
        visibility: 'public',
        title: 'New Calendar',
        startDate: '2026-07-01',
        endDate: '2026-07-31',
      })
    )
  })

  it('rejects non-creator write to existing calendar', async () => {
    await testEnv.withSecurityRulesDisabled(async (ctx) => {
      await setDoc(doc(ctx.firestore(), 'calendars/cal-existing'), {
        id: 'cal-existing',
        creatorId: 'uid-creator',
        status: 'published',
        visibility: 'public',
        title: 'Original Title',
      })
    })

    const other = testEnv.authenticatedContext('uid-other')
    await assertFails(
      setDoc(doc(other.firestore(), 'calendars/cal-existing'), {
        title: 'Hacked Title',
      })
    )
  })

  it('rejects unauthenticated create of calendar', async () => {
    const unauth = testEnv.unauthenticatedContext()
    await assertFails(
      setDoc(doc(unauth.firestore(), 'calendars/cal-unauth'), {
        creatorId: 'uid-creator',
        title: 'Unauth Calendar',
      })
    )
  })
})

// ── slots collection ─────────────────────────────────────────────────────────

describe('slots collection', () => {
  it('allows unauthenticated read of any slot', async () => {
    await testEnv.withSecurityRulesDisabled(async (ctx) => {
      await setDoc(doc(ctx.firestore(), 'slots/slot-1'), {
        id: 'slot-1',
        calendarId: 'cal-public',
        creatorId: 'uid-creator',
        status: 'available',
        startUtc: '2026-07-01T09:00:00Z',
        endUtc: '2026-07-01T10:00:00Z',
      })
    })

    const unauth = testEnv.unauthenticatedContext()
    await assertSucceeds(
      getDoc(doc(unauth.firestore(), 'slots/slot-1'))
    )
  })

  it('allows creator to create slot with matching creatorId', async () => {
    const creator = testEnv.authenticatedContext('uid-creator')
    await assertSucceeds(
      setDoc(doc(creator.firestore(), 'slots/slot-new'), {
        id: 'slot-new',
        calendarId: 'cal-public',
        creatorId: 'uid-creator',
        status: 'available',
        startUtc: '2026-07-01T11:00:00Z',
        endUtc: '2026-07-01T12:00:00Z',
      })
    )
  })

  it('rejects non-creator write to slot', async () => {
    await testEnv.withSecurityRulesDisabled(async (ctx) => {
      await setDoc(doc(ctx.firestore(), 'slots/slot-existing'), {
        id: 'slot-existing',
        calendarId: 'cal-public',
        creatorId: 'uid-creator',
        status: 'available',
        startUtc: '2026-07-01T13:00:00Z',
        endUtc: '2026-07-01T14:00:00Z',
      })
    })

    const other = testEnv.authenticatedContext('uid-other')
    await assertFails(
      setDoc(doc(other.firestore(), 'slots/slot-existing'), {
        status: 'booked',
      })
    )
  })
})

// ── bookings collection ──────────────────────────────────────────────────────

describe('bookings collection', () => {
  it('rejects all direct client writes to bookings (Admin SDK only)', async () => {
    // Even the creator cannot write directly — bookings go through Admin SDK
    const creator = testEnv.authenticatedContext('uid-creator')
    await assertFails(
      setDoc(doc(creator.firestore(), 'bookings/booking-1'), {
        id: 'booking-1',
        creatorId: 'uid-creator',
        bookerUid: 'uid-booker',
        status: 'confirmed',
      })
    )
  })

  it('rejects read of booking by non-creator non-booker', async () => {
    await testEnv.withSecurityRulesDisabled(async (ctx) => {
      await setDoc(doc(ctx.firestore(), 'bookings/booking-2'), {
        id: 'booking-2',
        creatorId: 'uid-creator',
        bookerUid: 'uid-booker',
        status: 'confirmed',
      })
    })

    const other = testEnv.authenticatedContext('uid-other')
    await assertFails(
      getDoc(doc(other.firestore(), 'bookings/booking-2'))
    )
  })
})

// ── invites collection ───────────────────────────────────────────────────────

describe('invites collection', () => {
  it('rejects authenticated write to invites', async () => {
    const user = testEnv.authenticatedContext('uid-user')
    await assertFails(
      setDoc(doc(user.firestore(), 'invites/inv-1'), {
        email: 'test@example.com',
        token: '123456',
        used: false,
      })
    )
  })

  it('rejects authenticated read of invites', async () => {
    await testEnv.withSecurityRulesDisabled(async (ctx) => {
      await setDoc(doc(ctx.firestore(), 'invites/inv-existing'), {
        email: 'test@example.com',
        token: '123456',
        used: false,
      })
    })

    const user = testEnv.authenticatedContext('uid-user')
    await assertFails(
      getDoc(doc(user.firestore(), 'invites/inv-existing'))
    )
  })
})
